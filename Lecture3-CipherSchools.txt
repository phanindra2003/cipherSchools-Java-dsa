package corejava;

public class primitiveDatatypes {

	public static void main(String[] args) {
		int x = 2;
		int y = 5;
		System.out.println("x is :"+x);
		System.out.println("y is :"+y);
		
		byte b=10;
		System.out.println("B is :"+b);
		
		byte b1=(byte)129;
		System.out.println("B1 is :"+b1);
		
		int i=b;
		System.out.println("I is :"+ i);
	}
	

}
